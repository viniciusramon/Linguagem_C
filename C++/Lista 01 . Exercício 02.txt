/* desenvolver um programa que
 calcule a média aritmética de
 duas notas lidas
*/

#include <stdio.h>

main() {
       
    float n1,n2,media;
    
    printf("Digite a primeira nota: ");
    scanf("%f",&n1);
    
    printf("Digite a segunda nota: ");
    scanf("%f",&n2);
    
    media = (n1 + n2) / 2;
    
    printf("Media: %.2f\n\n",media);
    
    if (media >= 6) {
         printf("APROVADO\n");     
    } 
    else {
         printf("REPROVADO\n");  
           
           }
              
              
     
    system("pause");
    
    
    
       
       }
